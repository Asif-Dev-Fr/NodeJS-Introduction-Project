index.js : 
    // Send information from the client to the server :
    app.use(express.json({limit: '1mb'}));
    // Data sont envoyés depuis le client au server (post)
    app.post('/api', (request, response) =>{
        console.log('Server side :')
        // le console.log suivant vient du client side, on envoit le data qui est récupérer dans le body de la request : (body: JSON.stringify(data),) : 
        console.log(request.body);
        response.json({
            statut: 'success',
            latitude: request.body.lat,
            longitude: request.body.lon,
        });
    });
